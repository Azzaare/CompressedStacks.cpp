92 Problem<T, D>::Problem(std::string fileName) : mIndex(0), mContext(nullptr) {
93  mInput.open(fileName, std::ifstream::in);
94 
95  std::vector<std::string> parameters = readHeader();
96 
97  int p, b, n = 0;
98  bool foundP = false;
99  bool foundBuffer = false;
100 
101  for (unsigned int i = 0; i < parameters.size(); i = i + 2) {
102  if (parameters[i].compare("n") == 0)
103  n = stoi(parameters[i + 1]);
104  if (parameters[i].compare("p") == 0) {
105  foundP = true;
106  p = stoi(parameters[i + 1]);
107  }
108  if (parameters[i].compare("b") == 0) {
109  foundBuffer = true;
110  b = stoi(parameters[i + 1]);
111  }
112  }
113 
114  if (foundBuffer && !foundP)
115  throw("Problem<T,D>::Problem(std::string fileName), wrong header format ");
116  if (!foundP)
117  mStack = std::shared_ptr<Stack<T, D>>(
118  new NormalStack<T, D>()); // space not provided, normal stack
119  else // space was provided, compressed stack
120  {
121  if (!foundBuffer)
122  b = 0;
123  mStack = std::shared_ptr<Stack<T, D>>(
124  new CompressedStack<T, D>(n, p, b, mContext));
125  }
126 }
